```python
def on_chat():
    agent.teleport_to_player()
    agent.move(FORWARD, 1)
    for x in range(4):
        agent.move(UP, 1)
        for y in range(4):
            for z in range(9):
                agent.move(FORWARD, 2)
                agent.place(UP)
            agent.turn(LEFT_TURN)
player.on_chat("build", on_chat)
```
